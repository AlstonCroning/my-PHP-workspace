CREATE TABLE Employee
(
	ssn VARCHAR(11) PRIMARY KEY,
	lastname VARCHAR(40) NOT NULL,
	firstname VARCHAR(30),
	department_code INT(3) NOT NULL,
	annual_salary NUMERIC(8,2),
	hire_date DATE NOT NULL
);
